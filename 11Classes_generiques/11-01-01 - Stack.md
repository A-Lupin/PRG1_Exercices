# Stack

Écrire l'implémentation d'une pile générique.

Une pile contient des `data` stockées sous la forme d'un `vector`.

Le code ci-après doit produire le résulat indiqué

~~~cpp
int main() {
   Stack<int> s;

   s.display();
   cout << endl;

   int i = 1;
   while (s.size() < 10) {
      s.push(i*=2);
   }

   cout << "top  : " << s.top() << endl;
   cout << "size : " << s.size() << endl;
   cout << endl;

   s.display();
   cout << endl;

   while (not s.empty()) {
      s.pop();
   }

   s.display();
   cout << endl;
}
~~~

~~~
size : 0
data : []

top  : 1024
size : 10

size : 10
data : [2, 4, 8, 16, 32, 64, 128, 256, 512, 1024]

size : 0
data : []
~~~

<summary>Solution</summary>

~~~cpp
template <typename T>
class Stack {
   std::vector<T> data;
public:
   Stack() : data{} {}

   // méthode définie plus bas
   void display() const;

   // méthodes triviales définies en ligne
   bool empty() const { return data.empty();}
   size_t size() const { return data.size(); }
   void push(const T& v){data.push_back(v);}
   void pop(){data.pop_back();}
   const T& top() const {return data.back();}
};

template <typename T>
void Stack<T>::display() const {
   cout << "size : " << size() << endl;
   cout << "data : ";

   cout << "[";
   for (size_t i = 0; i < size(); ++i) {
      if(i != 0) cout << ", ";
      cout << data[i];
   }
   cout << "]" << endl;
}
~~~
